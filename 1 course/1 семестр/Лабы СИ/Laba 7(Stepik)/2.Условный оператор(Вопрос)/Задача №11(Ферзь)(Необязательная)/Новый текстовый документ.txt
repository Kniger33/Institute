#include <stdio.h>
#include <stdlib.h>
#include <math.h>

int main () {

    int x1, y1, x2, y2;

    scanf("%d%d", &y1, &x1);
    scanf("%d%d", &y2, &x2);

    if ((fabs(x1 - x2) == fabs(y1 - y2)) || ((x1 == x2) || (y1 == y2))){
            printf("YES");
    }
    else printf("NO");

    return 0;
}
