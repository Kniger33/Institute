#include <stdio.h> 
#include <stdlib.h> 
#include <string.h> 

double pow(double a, int n){ 
    if (n == 0){ 
        return 1;
    }
    if (n % 2 == 0){ 
        return pow(a * a, n / 2);
    }
    else {
        return a * pow(a, n - 1);
    }
}
 

int main() 
{ 
    double a; 
    int n; 
    
    scanf("%lf%d", &a, &n); 
    printf("%lf\n", pow(a, n));
     
    return 0; 
}
