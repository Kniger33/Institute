#include <stdio.h>
#include <stdlib.h>
#include <math.h>

int main () {

    int N, K, I, R, i, j, k;


    scanf("%d%d", &N, &K);
    int mas[N];

    for (i = 0; i < N; i++) {
        mas[i] = i + 1;
    }

    for(i = 0; i < K; i++) {
        scanf("%d%d", &I, &R);
        for(j = 0; j < N; j++) {
            for(k = I - 1; k < R ; k++) {
                mas[k] = 0;
            }
        }
    }
    for (i = 0; i < N; i++) {
        if(mas[i] > 0){
            printf("I");
        }
        else {
            printf(".");
        }
    }
    return 0;
}
