#include <stdio.h> 
#include <stdlib.h> 
#include <math.h> 

int main() { 
    int m, n, i, j, i1, j1; 
    
    scanf("%d%d", &n, &m); 
    
    int a[n][m]; 
    int b[m][n]; 
    
    for (i = 0; i < n; i++){ 
        for (j = 0; j < m; j++){ 
            scanf("%d", &a[i][j]); 
        } 
    } 
    for (i = 0; i < n; i++){ 
        for (j = 0; j < m; j++){ 
            b[j][n - 1 - i] = a[i][j]; 
        } 
    } 
    for (i = 0; i < m; i++){ 
        for (j = 0; j < n; j++){ 
            printf("%d ", b[i][j]); 
        } 
        printf("\n"); 
    } 
    return 0; 
}
