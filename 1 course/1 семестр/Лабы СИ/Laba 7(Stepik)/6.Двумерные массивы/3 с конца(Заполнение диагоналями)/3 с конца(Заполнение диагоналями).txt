#include <stdio.h> 
#include <stdlib.h> 
#include <string.h> 

int main() 
{ 
    int n, m, i, j, row = 0, current = 0, begin = 0, cnt = 1; 
    
    scanf("%d%d", &n, &m); 
    int mas[n][m]; 
    
    for (i = 0; i < n; i++) { 
        if (i > 0){ 
            begin = current - 1; 
        }
        else { 
            begin = 0; 
        }
        for (current = begin; current < m; current++) { 
            row = i; 
            for (int column = current; column >= 0; column--) { 
                if (row < n) { 
                    mas[row][column] = cnt; 
                    cnt++; 
                    row++; 
                } 
                else {
                    break;
                }
            } 
        } 
    } 
    for (i = 0; i < n; i++) { 
        for (j = 0; j < m; j++) { 
            if (1 % 10 == 1) {
                printf(""); 
            }
            else if (mas[i][j] % 100 == mas[i][j]) {
                printf("  "); 
            }
            else if (mas[i][j] % 1000 == mas[i][j]) {
                printf(" "); 
            }
            printf("%4d", mas[i][j]); 
        } 
        printf("\n"); 
    } 
    
    return 0; 
}
