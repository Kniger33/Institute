#include <stdio.h>
#include <stdlib.h>

int main() {
    
    int N = 10000, max = 0, zero = 0, count = 1, pred;
    
    while (N != zero){
        pred = N;
        scanf("%d", &N);
        
        if ( N == pred){
            count++;
        }
        else if(N != pred ) {
            count = 1;
        }
        
        else if (count > max) {
            max = count;
            count = 1;
        }
        if (count > max) {
            max = count;
        }
    }
    if (max <= 0){
        printf("1");
    }
    else {
        printf("%d", max);
    }
    return 0;
}
