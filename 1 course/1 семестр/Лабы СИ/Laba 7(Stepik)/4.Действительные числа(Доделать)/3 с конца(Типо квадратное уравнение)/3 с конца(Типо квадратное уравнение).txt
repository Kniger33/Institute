#include <stdio.h>
#include <stdlib.h>
#include <math.h>

int main() {

    double a, b, c;
    double x1, x2, D;

    scanf("%lf%lf%lf", &a, &b, &c);
    
    if (a == 0 && b == 0 && c == 0){
        printf("3");
    }
    else if (a == 0 && b == 0 && c!= 0){
        printf("0");
    }
    else if (a == 0 && b!= 0 && c != 0) {
        printf("1 %lf", (-c / b));
    }
    else { 
    
        D = (b * b) - (4 * a * c);

        if (D > 0){
            x1 = (-b + sqrt(D)) / (2 * a);
            x2 = (-b - sqrt(D)) / (2 * a);
            if (x1 > x2){
                double tmp;
                tmp = x1;
                x1 = x2;
                x2 = tmp;
            }
            printf("2 %lf %lf", x1, x2);
        }

        else if (D == 0){
            x1 = (-b + sqrt(D)) / (2 * a);
            printf("1 %lf", x1);
        }
        else if (D < 0){
            printf("0");
        }  
    }
   
    
    return 0;
}
