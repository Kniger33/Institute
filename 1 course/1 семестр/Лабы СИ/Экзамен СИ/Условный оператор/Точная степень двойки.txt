#include <stdio.h>
#include <math.h>

int main() {
    int N, y = 0, x = 0;

    scanf("%d", &N);

    while(N > 1) {
        if(N % 2 == 0){
            x++;
        }
        N /= 2;
        y++;
    }
    if(x == y){
        printf("YES");
    }
    else {
        printf("NO");
    }
  return 0;
}
