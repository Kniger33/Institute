#include <stdio.h>
#include <math.h>

int main() {

    int a, b, c;
    scanf("%d%d%d", &a, &b, &c);

    if (c < a * b) {
        if (c % b == 0 || c % a == 0 ) {
            printf("YES");
        }
        else {
            printf("NO");
        }
    }
    else {
        printf("NO");
    }
    return 0;
}
