#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int main () {

    char str[1000];
    int i, j, length, count = 0;
    scanf("%s", &str);

    length = strlen(str);

    for(i = 0, j = length - 1; i < (length / 2); i++, j--){
            if (str[i] == str[j]){
                count ++;
            }
    }
    if (count == (length / 2)){
        printf("yes");
    }
    else {
        printf("no");
    }

    return 0;
}
