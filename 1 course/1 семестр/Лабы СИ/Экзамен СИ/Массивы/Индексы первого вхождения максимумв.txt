#include <stdio.h> 

int main() { 
    
    int n, m, i, j; 
    scanf ("%d %d", &n, &m); 
    int a[n][m]; 
    
    for (i = 0; i < n; i++) { 
        for (j = 0; j < m; j++) { 
            scanf ("%d", &a[i][j]); 
        } 
    } 
    
    int max = a[0][0], max_i = 0, max_j = 0; 
    for (i = 0; i < n; i++) { 
        for (j = 0; j < m; j++) { 
            if (a[i][j] > max){ 
                max = a[i][j]; 
                max_i = i; 
                max_j = j; 
            } 
        } 
    } 

    printf ("%d", max_i); 
    printf (" "); 
    printf ("%d", max_j); 
    
    return 0; 
}
