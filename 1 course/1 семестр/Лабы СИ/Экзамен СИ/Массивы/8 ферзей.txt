#include <stdio.h>
#include <stdlib.h>
#include <math.h>

int main () {

    int i, j, sum = 0;
    int mas[16];

    for(i = 0; i < 16; i++) {
        scanf("%d", &mas[i]);
    }

    for(i = 0; i < 16; i+= 2) {
        for(j = i + 2; j < 16; j+= 2) {
            if ( ((mas[i] - mas[j]) == (mas[i + 1] - mas[j + 1])) || ((mas[i] - mas[j]) == (mas[j + 1] - mas[i + 1])) || (mas[i] == mas[j]) || (mas[i + 1] == mas[j + 1]) ){
                sum++;
            }
        }
    }

    if (sum > 0){
        printf("YES");
    }
    else {
        printf("NO");
    }
    return 0;
}
