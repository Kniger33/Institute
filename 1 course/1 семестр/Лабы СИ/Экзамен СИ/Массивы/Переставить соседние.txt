#include <stdio.h>
     
int main () {
    int N;
    scanf("%d", &N);
    int mas [N];
     
    for (int i = 0; i < N; i++){
        scanf("%d", &mas[i]);
    }
     
    for (int i = 0; i < N - 1 ; i += 2){
        int swap = mas[i];
        mas[i] = mas[i + 1];
        mas[i + 1] = swap;
    }
     
    for(int i = 0; i < N; i++){
        printf("%d ", mas[i]);
    }
    
    return 0;
}
