#include <stdio.h>
     
int main () {
    int N;
    scanf("%d", &N);
    int max = N - 1;
    int mas [N];
     
    for (int i = 0; i < N; i++){
        scanf("%d", &mas[i]);
    }
     
    for (int i = 0; i < N/2; i++){
        int swap = mas[i];
        mas [i] = mas[max];
        mas[max] = swap;
        max--;
    }
     
    for(int i = 0; i < N; i++){
        printf("%d ", mas[i]);
    }
    
    return 0;
}
