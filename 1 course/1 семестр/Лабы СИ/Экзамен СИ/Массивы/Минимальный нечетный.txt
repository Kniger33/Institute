#include <stdio.h>
#include <limits.h>


int main () {

    int N, y = 0, i;
    int min = INT_MAX;
    scanf("%d", &N);
    int mas [N];


    for (i = 0; i < N; i++){
        scanf("%d", &mas[i]);
        if (mas[i - 1] < mas[i] && mas[i - 1] < min && mas[i - 1] %2 != 0) {
            min = mas[i - 1];
        }
    }

    for (i = 0; i < N; i++) {

        if (mas[i] % 2 != 0) {

            if (mas[i] < min) {

                min = mas[i];

            }
        }
        else {
            y++;
        }

    }

    if (y == N) {
        printf("0");
    }
    else {
        printf("%d", min);
    }

    return 0;
}
