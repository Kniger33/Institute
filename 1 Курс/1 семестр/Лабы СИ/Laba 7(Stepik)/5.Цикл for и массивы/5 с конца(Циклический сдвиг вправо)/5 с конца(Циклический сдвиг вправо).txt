#include <stdio.h>
     
int main () {
    int N;
    scanf("%d", &N);
    int mas [N];
     
    for (int i = 0; i < N; i++){
        scanf("%d", &mas[i]);
    }
    int last = mas[N - 1];
     
    for (int i = N - 1; i > 0; i--){
        mas[i] = mas[i - 1];
    }
     
    mas[0] = last;
     
    for(int i = 0; i < N; i++){
        printf("%d ", mas[i]);
    }
    
    return 0;
}
