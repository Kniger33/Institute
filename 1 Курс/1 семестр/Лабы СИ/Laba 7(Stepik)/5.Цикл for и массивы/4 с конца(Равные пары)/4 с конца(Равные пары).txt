#include <stdio.h>
     
int main () {
    int N, i, j, count = 0;
    scanf("%d", &N);
    int mas [N];
     
    for (i = 0; i < N; i++) {
        scanf("%d", &mas[i]);
    }
    for (i = 0; i < N; i++){
        for(j = i + 1; j < N; j++){
            if (mas[i] == mas[j]){
                count++;
            }
        }
    }

    printf("%d ", count);
    
    return 0;
}
