#include <stdio.h>
#include <stdlib.h>
#include <math.h>

int main() {

    double a, b, c;
    double x1, x2, D;

    scanf("%lf%lf%lf", &a, &b, &c);

    D = (b * b) - (4 * a * c);
    if (D == 0){
        x1 = (-b + sqrt(D)) / (2 * a);
        printf("%lf", x1);
    }
    else if (D > 0){
        x1 = (-b + sqrt(D)) / (2 * a);
        x2 = (-b - sqrt(D)) / (2 * a);
        if (x1 > x2){
        double tmp;
        tmp = x1;
        x1 = x2;
        x2 = tmp;
    }
        printf("%lf %lf", x1, x2);
    }
    else {
        printf("\n");
    }

    return 0;
}
