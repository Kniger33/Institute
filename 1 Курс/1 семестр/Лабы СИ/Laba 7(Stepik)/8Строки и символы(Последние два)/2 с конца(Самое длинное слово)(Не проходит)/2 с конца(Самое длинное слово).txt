#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int main () {

    char str[100];
    int i, j, max = 0, count = 0, dlina, id = 0;

    gets(str);
    dlina = strlen(str);

    for(i = 0; i < dlina; i++){
        if(str[i] != ' '){
            count++;
        }
        else {
            if (count > max){
                max = count;
                id = i - count;
            }
            count = 0;
        }
        if (count > max){
            max = count;
            id = i - count;
        }
    }
    max += id;
    for(i = id; i < max + 1; i++){
        printf("%c", str[i]);
    }

    return 0;
}
