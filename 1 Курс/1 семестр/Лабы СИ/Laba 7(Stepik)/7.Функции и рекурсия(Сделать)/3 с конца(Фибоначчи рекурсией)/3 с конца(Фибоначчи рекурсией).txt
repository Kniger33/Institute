#include <stdio.h>

int sum (int first, int second, int n)
{
    int fib;
    if (n != 0)
    {
        fib = first + second;
        first = second;
        second = fib;
        --n;
        sum (first, second, n);
    } else
    {
        printf("%d", second);
    }
}
int main()
{
    int first = 1, second = 1, n;
    scanf("%d", &n);
    if ((n == 1) || (n == 2))
    {
        printf("1");
    } else
    {
        n -= 2;
        sum (first, second, n);
    }
    return 0;
}
