#include <stdio.h>
#include <stdlib.h>


int main()
{
    int **mas;
    int i, j, n, *m;

    printf("Num of strings: ");
    scanf("%d", &n);

    mas = (int**)malloc(n * sizeof(int*));
    m = (int*)malloc(n * sizeof(int));

    for (i = 0; i < n; i++) {

        printf("Num of columns in string %d: ", i);
        scanf("%d", &m[i]);
        mas[i] = (int*)malloc(m[i] * sizeof(int));

        for (j = 0; j < m[i]; j++) {
            printf("mas[%d][%d]: ", i, j);
            scanf("%d", &mas[i][j]);
        }
    }

    for (i = 0; i < n; i++) {
        for (j = 0; j < m[i]; j++) {
            printf("%d ", mas[i][j]);
        }
        printf("\n");
    }


    for (i = 0; i < n; i++) {
        free(mas[i]);
    }
    free(mas);
    free(m);

    return 0;

}
