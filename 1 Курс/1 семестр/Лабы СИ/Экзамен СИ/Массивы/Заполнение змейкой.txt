                for(j = m - 1; j >= 0; j--){
                    mas[i][j] = x;
                    x++;
                }
            }
        }


        printf("\n");
        //printf(" ");

        for (i = 0; i < n; i++){
            for (j = 0; j < m; j++){

                    printf("  ");

                }

            }

        }
