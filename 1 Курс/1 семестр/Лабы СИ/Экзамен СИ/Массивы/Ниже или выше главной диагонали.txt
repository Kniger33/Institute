#include <stdio.h>
#include <math.h>

int main () {

    int N;

    scanf("%d", &N);
    int mas [N][N];

    int i, j;

    for (i = 0; i < N; i++) {
        for (j = 0; j < N; j++) {
            scanf("%d", &mas[i][j]);
        }
    }

    int k;
    scanf("%d", &k);

    for (i = 0; i < N; i++) {
        for (j = 0; j < N; j++) {
            if (i - j - k == 0) {
                printf("%d ", mas[i][j]);
            }
        }
    }

    

    return 0;
}
