d_s segment 
mas1 db 5 dup ('0')  ;резервируем 5 однобайтовых ячеек памяти для  
  			   ;массива из 5 элементов
a db 'ne uporyadochen$',13,10,'$' 
b db 'uporyadochen$',13,10,'$'
d_s ends 

c_s segment 
assume cs:c_s,ds:d_s 
begin:
	mov ax, d_s 
	mov ds, ax   ;записываем начало сегмента данных
	mov si, 0      ;обнуляем индексный регистр
	mov cx, 5     ;задаём количество повторений цикла
m1: 
	mov ah, 01h  ;функция ввода с клавиатуры
	int 21h          ;прерывание для ввода
	add al,30h    ;к каждому введённому элементу прибавляем по 30
                               ;чтобы получить именно это число в таблице 
                               ;ASCII
	mov  mas1[si], al    ;записываем полученный элемент  в массив 
	inc si                       ;инкремируем регистр si
loop m1                             ;повторяем цикл для заполнения массива


	mov si, 0                 ;в регистр si записываем 0 индекс массива с 
                                          ;которого будем проверять упорядоченность
	mov cx,4                 ;задаём количество повторений для цикла
repeat:                               ;организуем цикл для проверки массива на 
                                          ;упорядоченность
	mov bl, mas1[si]     ;в регистр bl записываем элемент массива 
                                          ;находящийся под эти  индексом
	mov dl, mas1[si+1] ;в регистр dl записываем следующий элемент
                                           ; за тем, что записали в регистр bl
	cmp bl, dl                 ;сравниваем  попарно элементы массива 
                                           ;между собой
	jb m2                        ;если bl<dl перейти на метку m2
	mov ah, 0h                ;функция установки экрана
	mov al, 2h                 ;установка видеорежима
	int 10h                       ;прерывание для данной функции
	mov ah, 9h                ;функция вывода на экран строки
	mov dx, offset a        ;непосредственный вывод строки
	mov bh, 0                  ;установка экранной страницы
	int 21h                       ;прерывание для вывода на экран
jmp exit                               ;безусловный переход на метку exit

m2:
	inc si                        ;инкремируем регистр si
loop repeat                        ;повторяем цикл на проверку упорядоченности
	mov ah, 0h              ;функция установки экрана
	mov al, 2h               ;установка видеорежима
	int 10h	            ;прерывание для данной функции
	mov ah, 9h              ;функция вывода на экран строки
	mov dx, offset b      ;непосредственный вывод строки
	mov bh, 0                ;установка экранной страницы
	int 21h                     ;прерывание для вывода на экран

exit:
	mov ah, 4ch             ;корректное завершение работы программы
	int 21h
	c_s ends
	end begin
